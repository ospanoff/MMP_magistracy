\begin{frame}
\frametitle{Why fingerprint?}
\begin{itemize}
    \item Efficient mechanism to establish the perceptual equality of two audio objects
    \item Advantages
    \begin{itemize}
        \item[-] Reduced memory/storage requirements
        \item[-] Efficient comparison
        \item[-] Efficient search
    \end{itemize}
\end{itemize}
\end{frame}
